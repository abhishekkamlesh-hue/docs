---
title: "Update User Script Data"
api: "POST /updateUserSpecificScriptData"
description: "Update user-specific customizations for a script"
---

##  Overview

Modify user-specific script configurations without affecting the base script template.

---

##  Authentication

<ParamField header="Cookie" type="string" required>
  Your API key for authentication
</ParamField>

<ParamField header="Accept" type="string">
  application/json, text/plain, */*
</ParamField>

<ParamField header="Origin" type="string">
  Origin of the request (e.g., https://vb-seamless-dev-backend.oriserve.com/)
</ParamField>

<ParamField header="Content-Type" type="string" required>
  application/json
</ParamField>

<ParamField header="User-Agent" type="string">
  Browser or client user agent
</ParamField>

---

##  Request Body

<ParamField body="script_id" type="string" required>
  Script ID (MongoDB ObjectId)
  ```
  507f1f77bcf86cd799439012
  ```
</ParamField>

<ParamField body="user_id" type="string" required>
  User ID making the changes
  ```
  507f1f77bcf86cd799439011
  ```
</ParamField>

<ParamField body="user_specific_data" type="object" required>
  Custom script data for this user
</ParamField>

### User Specific Data Fields:

<ParamField body="user_specific_data.custom_prompts" type="array">
  Custom conversation prompts
  ```json
  [
    {
      "trigger": "greeting",
      "text": "Hello from {company_name}"
    }
  ]
  ```
</ParamField>

<ParamField body="user_specific_data.variables" type="object">
  Dynamic variables
  ```json
  {
    "company_name": "Acme Corp",
    "support_email": "support@acme.com"
  }
  ```
</ParamField>

<ParamField body="user_specific_data.custom_actions" type="array">
  Custom actions/webhooks
  ```json
  [
    {
      "event": "call_ended",
      "webhook_url": "https://api.example.com/call-ended"
    }
  ]
  ```
</ParamField>

### Example Request:
```json
{
  "script_id": "507f1f77bcf86cd799439012",
  "user_id": "507f1f77bcf86cd799439011",
  "user_specific_data": {
    "custom_prompts": [
      {
        "trigger": "greeting",
        "text": "Hello from Acme Corp, how can I assist you?"
      },
      {
        "trigger": "goodbye",
        "text": "Thank you for calling Acme Corp. Have a great day!"
      }
    ],
    "variables": {
      "company_name": "Acme Corp",
      "support_email": "support@acme.com",
      "business_hours": "9 AM to 6 PM IST"
    },
    "custom_actions": [
      {
        "event": "call_ended",
        "webhook_url": "https://api.example.com/call-ended",
        "method": "POST"
      }
    ]
  }
}
```

## Code Examples

### JavaScript:
```javascript
const axios = require('axios');

async function updateUserScriptData(scriptId, userId, customData) {
  const response = await axios.post(
    'https://vb-seamless-dev-backend.oriserve.com/updateUserSpecificScriptData',
    {
      script_id: scriptId,
      user_id: userId,
      user_specific_data: customData
    },
    {
      headers: {
        'x-api-key': 'YOUR_API_KEY',
        'Content-Type': 'application/json'
      }
    }
  );
  
  console.log('âœ… User script data updated');
  return response.data.data;
}

// Example: Customize greetings
updateUserScriptData(
  '507f1f77bcf86cd799439012',
  '507f1f77bcf86cd799439011',
  {
    custom_prompts: [{
      trigger: 'greeting',
      text: 'Hello from Acme Corp!'
    }],
    variables: {
      company_name: 'Acme Corp'
    }
  }
);
```
