title: "Redial Campaign Calls"
api: "POST /redialCampaignCalls"
description: "Retry failed or unanswered calls from a campaign"
---

##  Overview

Manually trigger redial for specific calls or all failed/unanswered calls in a campaign.

---

##  Authentication

<ParamField header="Cookie" type="string" required>
  Your API key for authentication
</ParamField>

<ParamField header="Accept" type="string">
  application/json, text/plain, */*
</ParamField>

<ParamField header="Origin" type="string">
  Origin of the request (e.g., https://vb-seamless-dev-backend.oriserve.com/)
</ParamField>

<ParamField header="Content-Type" type="string" required>
  application/json
</ParamField>

<ParamField header="User-Agent" type="string">
  Browser or client user agent
</ParamField>

---

##  Request Body

<ParamField body="campaign_id" type="string" required>
  Campaign ID
  ```
  507f1f77bcf86cd799439013
  ```
</ParamField>

<ParamField body="call_uuids" type="array">
  Specific call UUIDs to redial (optional)
  ```json
  ["call-uuid-abc123", "call-uuid-def456"]
  ```
</ParamField>

<ParamField body="redial_criteria" type="object">
  Criteria for bulk redial
</ParamField>

### Redial Criteria Fields:

<ParamField body="redial_criteria.statuses" type="array">
  Call statuses to redial
  ```json
  ["no-answer", "busy", "failed"]
  ```
</ParamField>

<ParamField body="redial_criteria.max_previous_attempts" type="integer">
  Maximum previous attempts (default: 3)
  ```
  2
  ```
</ParamField>

<ParamField body="redial_criteria.exclude_dnc" type="boolean">
  Exclude DNC numbers (default: true)
  ```
  true
  ```
</ParamField>

### Example Request 1 (Specific Calls):
```json
{
  "campaign_id": "507f1f77bcf86cd799439013",
  "call_uuids": [
    "call-uuid-abc123",
    "call-uuid-def456"
  ]
}
```

### Example Request 2 (Bulk Redial):
```json
{
  "campaign_id": "507f1f77bcf86cd799439013",
  "redial_criteria": {
    "statuses": ["no-answer", "busy"],
    "max_previous_attempts": 2,
    "exclude_dnc": true
  }
}
```

## Code Examples

### cURL:
```bash
curl -X POST "https://vb-seamless-dev-backend.oriserve.com/redialCampaignCalls" \
  -H "x-api-key: YOUR_API_KEY" \
  -H "Content-Type: application/json" \
  -d '{
    "campaign_id": "507f1f77bcf86cd799439013",
    "redial_criteria": {
      "statuses": ["no-answer", "busy"],
      "max_previous_attempts": 2
    }
  }'
```

### JavaScript:
```javascript
const axios = require('axios');

async function redialCampaignCalls(campaignId, criteria) {
  const response = await axios.post(
    'https://vb-seamless-dev-backend.oriserve.com/redialCampaignCalls',
    {
      campaign_id: campaignId,
      redial_criteria: criteria
    },
    {
      headers: {
        'x-api-key': 'YOUR_API_KEY',
        'Content-Type': 'application/json'
      }
    }
  );
  
  const { total_calls_to_redial, calls_scheduled } = response.data.data;
  console.log(`âœ… Scheduled ${calls_scheduled} of ${total_calls_to_redial} calls for redial`);
  
  return response.data.data;
}

// Example: Redial all unanswered calls
redialCampaignCalls('507f1f77bcf86cd799439013', {
  statuses: ['no-answer'],
  max_previous_attempts: 2,
  exclude_dnc: true
});
```

## Use Cases

### 1. Smart Retry Logic:
```javascript
async function retryFailedCalls(campaignId) {
  // First attempt: busy and no-answer
  await redialCampaignCalls(campaignId, {
    statuses: ['busy', 'no-answer'],
    max_previous_attempts: 1
  });
  
  // Wait 2 hours, then retry again
  setTimeout(async () => {
    await redialCampaignCalls(campaignId, {
      statuses: ['no-answer'],
      max_previous_attempts: 2
    });
  }, 7200000);
}
```

### 2. Specific Number Retry:
```javascript
async function retrySpecificNumbers(campaignId, callUuids) {
  const response = await axios.post(
    'https://vb-seamless-dev-backend.oriserve.com/redialCampaignCalls',
    {
      campaign_id: campaignId,
      call_uuids: callUuids
    },
    {
      headers: { 'x-api-key': 'YOUR_API_KEY' }
    }
  );
  
  return response.data.data;
}
```
