##  Overview

Get detailed analytics on AI bot response times, latency breakdown, and performance trends.

## Code Examples

### cURL:
```bash
curl -X GET "https://vb-seamless-dev-backend.oriserve.com/botResTimeAnalytics?interval=day&start_date=2024-01-01T00:00:00Z&end_date=2024-01-31T23:59:59Z" \
  -H "x-api-key: YOUR_API_KEY"
```

### JavaScript:
```javascript
const axios = require('axios');

async function getBotResponseTimeAnalytics(filters) {
  const response = await axios.get(
    'https://vb-seamless-dev-backend.oriserve.com/botResTimeAnalytics',
    {
      params: {
        start_date: filters.start_date,
        end_date: filters.end_date,
        interval: filters.interval || 'day',
        campaign_id: filters.campaign_id
      },
      headers: {
        'x-api-key': 'YOUR_API_KEY'
      }
    }
  );
  
  const { overview, latency_breakdown } = response.data.data;
  
  console.log('\n Bot Response Time Analytics');
  console.log(`Average: ${overview.average_response_time}s`);
  console.log(`P95: ${overview.p95_response_time}s`);
  console.log(`\nðŸ“Š Latency Breakdown:`);
  console.log(`   Speech Recognition: ${latency_breakdown.speech_recognition.average}s (${latency_breakdown.speech_recognition.percentage_of_total}%)`);
  console.log(`   LLM Processing: ${latency_breakdown.llm_processing.average}s (${latency_breakdown.llm_processing.percentage_of_total}%)`);
  console.log(`   Text-to-Speech: ${latency_breakdown.text_to_speech.average}s (${latency_breakdown.text_to_speech.percentage_of_total}%)`);
  
  return response.data.data;
}

// Example
getBotResponseTimeAnalytics({
  start_date: '2024-01-01T00:00:00Z',
  end_date: '2024-01-31T23:59:59Z',
  interval: 'day'
});
```

### 2. Optimization Insights:
```javascript
async function identifyBottlenecks() {
  const data = await getBotResponseTimeAnalytics({
    start_date: '2024-01-01T00:00:00Z'
  });
  
  const breakdown = data.latency_breakdown;
  
  // Find highest latency component
  const components = [
    { name: 'Speech Recognition', value: breakdown.speech_recognition.percentage_of_total },
    { name: 'LLM Processing', value: breakdown.llm_processing.percentage_of_total },
    { name: 'TTS', value: breakdown.text_to_speech.percentage_of_total }
  ];
  
  const bottleneck = components.sort((a, b) => b.value - a.value)[0];
  console.log(`ðŸŽ¯ Focus optimization on: ${bottleneck.name} (${bottleneck.value}%)`);
}
```
