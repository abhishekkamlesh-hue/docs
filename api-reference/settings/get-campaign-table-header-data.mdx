## Overview

Get the column configuration for campaign data tables including custom fields and display preferences.

```bash
  -H "x-api-key: YOUR_API_KEY"
```

### JavaScript (Build Dynamic Table):
```javascript
const axios = require('axios');

async function buildCampaignTable() {
  // Get table configuration
  const configResponse = await axios.get(
    'https://vb-seamless-dev-backend.oriserve.com/getCampaignTableHeaderData',
    {
      headers: { 'x-api-key': 'YOUR_API_KEY' }
    }
  );
  
  const config = configResponse.data.data;
  
  // Get campaign data
  const dataResponse = await axios.get(
    'https://vb-seamless-dev-backend.oriserve.com/fetchCampaignData',
    {
      headers: { 'x-api-key': 'YOUR_API_KEY' }
    }
  );
  
  const campaigns = dataResponse.data.data.campaigns;
  
  // Build table columns from config
  const visibleColumns = config.default_columns
    .filter(col => col.visible)
    .map(col => ({
      key: col.key,
      label: col.label,
      type: col.type,
      width: col.width,
      sortable: col.sortable
    }));
  
  console.log('\nüìä Campaign Table Configuration:');
  console.log('Columns:', visibleColumns.map(c => c.label).join(', '));
  console.log(`\nShowing ${campaigns.length} campaigns\n`);
  
  // Display table
  campaigns.forEach((campaign, index) => {
    console.log(`${index + 1}. ${campaign.campaignName}`);
    console.log(`   Status: ${campaign.status}`);
    console.log(`   Contacts: ${campaign.totalContacts}`);
    console.log(`   Success Rate: ${campaign.successRate || 0}%\n`);
  });
  
  return { config, campaigns };
}

// Build and display table
buildCampaignTable();
```

### React Component Example:
```jsx
import React, { useEffect, useState } from 'react';
import axios from 'axios';

function CampaignTable() {
  const [tableConfig, setTableConfig] = useState(null);
  const [campaigns, setCampaigns] = useState([]);
  
  useEffect(() => {
    async function loadData() {
      // Get table config
      const configRes = await axios.get(
        'https://vb-seamless-dev-backend.oriserve.com/getCampaignTableHeaderData',
        { headers: { 'x-api-key': 'YOUR_API_KEY' } }
      );
      setTableConfig(configRes.data.data);
      
      // Get campaigns
      const dataRes = await axios.get(
        'https://vb-seamless-dev-backend.oriserve.com/fetchCampaignData',
        { headers: { 'x-api-key': 'YOUR_API_KEY' } }
      );
      setCampaigns(dataRes.data.data.campaigns);
    }
    
    loadData();
  }, []);
  
  if (!tableConfig) return <div>Loading...</div>;
  
  const visibleColumns = tableConfig.default_columns.filter(col => col.visible);
  
  return (
    <table>
      <thead>
        <tr>
          {visibleColumns.map(col => (
            <th key={col.key} style={{ width: col.width }}>
              {col.label}
              {col.sortable && <span>‚¨ç</span>}
            </th>
          ))}
        </tr>
      </thead>
      <tbody>
        {campaigns.map(campaign => (
          <tr key={campaign._id}>
            {visibleColumns.map(col => (
              <td key={col.key}>{campaign[col.key]}</td>
            ))}
          </tr>
        ))}
      </tbody>
    </table>
  );
}

export default CampaignTable;
```
