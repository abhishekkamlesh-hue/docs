## Code Examples

### cURL:
```bash
curl -X GET "https://vb-seamless-dev-backend.oriserve.com/getConversationLogs/call-uuid-abc123" \
  -H "x-api-key: YOUR_API_KEY"
```

### JavaScript:
```javascript
const axios = require('axios');

async function getConversationLogs(callId) {
  const response = await axios.get(
    `https://vb-seamless-dev-backend.oriserve.com/getConversationLogs/${callId}`,
    {
      headers: {
        'x-api-key': 'YOUR_API_KEY'
      }
    }
  );
  
  const { conversation, sentiment_analysis } = response.data.data;
  
  console.log(`\nCall: ${callId}`);
  console.log(`Sentiment: ${sentiment_analysis.overall_sentiment}\n`);
  
  conversation.forEach(msg => {
    const speaker = msg.role === 'user' ? ' User' : 'ðŸ¤– AI';
    console.log(`${speaker}: ${msg.content}`);
  });
  
  return response.data.data;
}

// Example
getConversationLogs('call-uuid-abc123');
```

### Python:
```python
import requests

def get_conversation_logs(call_id):
    url = f"https://vb-seamless-dev-backend.oriserve.com/getConversationLogs/{call_id}"
    
    headers = {
        "x-api-key": "YOUR_API_KEY"
    }
    
    response = requests.get(url, headers=headers)
    
    if response.status_code == 200:
        data = response.json()['data']
        
        print(f"\nCall: {call_id}")
        print(f"Sentiment: {data['sentiment_analysis']['overall_sentiment']}\n")
        
        for msg in data['conversation']:
            speaker = " User" if msg['role'] == 'user' else "ðŸ¤– AI"
            print(f"{speaker}: {msg['content']}")
        
        return data
    else:
        print("Error:", response.json())
        return None

# Example
get_conversation_logs("call-uuid-abc123")
```

## Use Cases

### 1. Quality Assurance Review:
```javascript
async function reviewCallQuality(callId) {
  const logs = await getConversationLogs(callId);
  
  const qualityScore = {
    greeting: logs.conversation[0].content.includes('Hi'),
    response_time: logs.summary.avg_response_time < 2,
    outcome_achieved: logs.summary.outcome !== 'no_outcome',
    sentiment_positive: logs.sentiment_analysis.overall_sentiment === 'positive'
  };
  
  console.log('Quality Score:', qualityScore);
}
```

### 2. Training Data Collection:
```javascript
async function collectSuccessfulCalls() {
  const successfulCalls = await getCallRecordings({
    outcome: 'demo_scheduled'

  for (const call of successfulCalls) {

  }
}
```
